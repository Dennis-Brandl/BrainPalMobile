    **Step 4: Add reactivation method to WorkflowRunner (packages/engine/src/runner/workflow-runner.ts)**

    Add a new public method `reactivateSteps` that processes recovered steps based on their action type. This method handles three cases:

    ```typescript
    /**
     * Reactivate steps after crash recovery.
     * Routes each step to the correct execution entry point based on its recovery action:
     * - 'reactivate': step is in WAITING state, needs WAITING -> STARTING -> EXECUTING flow
     * - 're-execute': step is in EXECUTING state, needs executeExecutingPhase()
     * - 're-complete': step is in COMPLETING state, needs executeCompletingPhase() + onStepCompleted()
     */
    async reactivateSteps(
      workflowInstanceId: string,
      steps: Array<{ stepOid: string; stepInstanceId: string; action: 'reactivate' | 're-execute' | 're-complete' }>,
    ): Promise<void> {
      for (const { stepOid, stepInstanceId, action } of steps) {
        if (action === 'reactivate') {
          // Step is in WAITING -- re-enter normal activation from WAITING onward
          // Enqueue through event queue for serial processing
          await this.eventQueue.enqueue({
            type: 'STEP_ACTIVATED',
            stepInstanceId,
            workflowInstanceId,
            stepOid,
          });
        } else if (action === 're-execute') {
          // Step is in EXECUTING -- call executeExecutingPhase directly
          await this.reactivateExecutingStep(workflowInstanceId, stepOid, stepInstanceId);
        } else if (action === 're-complete') {
          // Step is in COMPLETING -- call executeCompletingPhase directly
          await this.reactivateCompletingStep(workflowInstanceId, stepOid, stepInstanceId);
        }
      }
    }
    ```

      stepOid: string,
      stepInstanceId: string,
    ): Promise<void> {

      if (!runnerState) return;
      const step = await this.config.stepRepo.getById(stepInstanceId);
      if (!step) return;

        if (!completingStep) return;

    Replace with the simplified version that consumes the rich `RecoveredWorkflowData`:

    ```typescript
    async function runRecovery() {
      try {
        const result = await recoverWorkflows(config);

        for (const recoveredData of result.recovered) {
          const { workflowInstanceId, runnerState, stepsToReactivate } = recoveredData;

          // Restore runner state (already built by crash-recovery.ts)
          runner.restoreWorkflowState(runnerState);

          // Get workflow for metadata
          const workflow = await config.workflowRepo.getById(workflowInstanceId);
          if (!workflow) continue;

          // Pre-populate parent cache
          if (workflow.parent_workflow_instance_id !== null) {
            parentCache.set(workflowInstanceId, workflow.parent_workflow_instance_id);
            // Child workflow -- restore into runner but don't add to UI
            continue;
          }
          parentCache.set(workflowInstanceId, null);

          // Parse spec for workflow name
          let name = 'Workflow';
          try {
            const spec = JSON.parse(workflow.specification_json);
            name = spec.local_id || spec.description || 'Workflow';
          } catch {
            // Use default
          }

          // Get step count for UI
          const steps = await config.stepRepo.getByWorkflow(workflowInstanceId);
          store.addActiveWorkflow(workflowInstanceId, workflow.master_workflow_oid, name, steps.length);

          // Reactivate frozen automated steps
          if (stepsToReactivate.length > 0) {
            runner.reactivateSteps(workflowInstanceId, stepsToReactivate).catch((err) => {
              console.warn(`Step reactivation failed for ${workflowInstanceId}:`, err);
            });
          }
        }
      } catch (err) {
        console.warn('Crash recovery failed:', err);
      }

      setIsReady(true);
    }
    ```

    - No more manual state machine/scheduler rebuilding (consumed from RecoveryResult)
    - Calls `runner.reactivateSteps()` for frozen automated steps
    - Fire-and-forget reactivation (don't block UI readiness on step processing)
